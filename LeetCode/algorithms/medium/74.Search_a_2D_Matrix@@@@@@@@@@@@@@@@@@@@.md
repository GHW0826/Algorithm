class Solution {
public:
    bool searchMatrix(vector<vector<int>>& matrix, int target) {
        bool answer = false;
        int row = matrix.size();
        int col = matrix[0].size();
        for (int i = 0; i < row; ++i) {
            if (target <= matrix[i][col - 1]) {
                if (find(matrix[i].begin(), matrix[i].end(), target) != matrix[i].end())
                    answer = true;
            }
        }
        return answer;
    }
};
